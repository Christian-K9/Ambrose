dns
	Server:
		sudo systemctl enable named
		sudo systemctl start named
		sudo nano /etc/named.conf
		
		config:
			listen-on port 53 { any; };
			allow-query	{ any; };

			zone "example.local" IN {
				type master;
				file "/var/named/example.local.db";
				allow-update { none; };
			};
	
			zone <ip_address_backwards-inaddr.arpa" IN {
				type master;
				file "/var/named/<ip-address.db";
				allow-update { none; };
			};

		sudo nano /var/named/ccdc.local.db
		
		config:
			$TTL 1D
			@	IN SOA ns1.ccdc.local. root.ccdc.local. (
	
					2025090401 ; serial
					1D	; refresh
					1H	; retry
					1W	; explore
					3H )	; minimum
				IN NS ns1.example.local.
			ns1	IN A	<ip-address>
			sev	IN A	<new-ip>

		sudo nano /var/named/<ip-address.db>
		
		config:
			$TTL 1D
			@	IN SOA ns1.ccdc.local. root.ccdc.local (
					2025090401 ; serial
					1D
					1H
					1W
					3H )
				IN NS ns1.ccdc.local.
			10	IN PTR ns1.ccdc.local.
`			20	IN PTR srv.ccdc.local.

		nano /etc/resolv.conf
			#add ip-address

		sudo chown named:namd /var/named/*.db
		sudo chmod 640 /var/named/*.db
		sudo systemctl status named
	
		sudo firewall-cmd --get-zone-of-interface <iface>
		sudo firewall-cmd --zone=<zone-name> --permanent --add-service=dns
		sudo firewall-cmd --zone=<zone-name> --permanent --add-port=53/tcp
		sudo firewall-cmd --zone=<zone-name> --permanent --add-port=53/udp
		sudo firewall-cmd --reload


	Client:
		dig @192.168.2.10 ns1.ccdc.local
		dig @192.168.2.10 google.com
